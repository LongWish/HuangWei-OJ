#include <stdio.h>
#include <string.h>
int cnt;
char input[250];
char str[250][250];
_Bool flag = 0; 
_Bool check(char *s){
    for(int i = 0; i < strlen(s); ++i)
    
        if(s[i] != '\n') return 1;
    return 0;
}
void solve(){
    _Bool evil = 0;
    for(int i = 0; i < cnt; ++i){
        int len = strlen(str[i]); 
        for(int j = 0; j < len; ++j){
            if(str[i][j] == '/' && str[i][j + 1] == '*' && !evil)
                str[i][j] = '\0', str[i][j + 1] = '\0', evil = 1;
                
            if(str[i][j] == '*' && str[i][j + 1] == '/' && evil)
                str[i][j] = '\0', str[i][j + 1] = '\0', evil = 0;
            if(evil) str[i][j] = '\0';
            if(str[i][j] == '/' && str[i][j + 1] == '/'){
                for(int k = j; k < len; ++k) str[i][k] = '\0';
            }
        }
    }
    
    _Bool is_start = 1;
    _Bool is_enter = 0; 
    for(int i = 0; i < cnt; ++i){
        if(!strlen(str[i])) continue; 
        if(check(str[i])){
            is_start = 0;
            is_enter = 0;
            printf("%s\n", str[i]);
        }
        else{
            if(is_start || is_enter) continue;
            printf("%s\n", str[i]);
            is_enter = 1;
        }
    }
}
int main(){
    while(gets(input) != NULL){
        if(strstr(input, "###") != NULL){
            if(flag == 0){
                flag = 1;
                cnt = 0;
            }
            else{
                flag = 0;
                solve();
            }
        }
        else strcpy(str[cnt++], input);
    }
    return 0^(0-0)^0; 
}
